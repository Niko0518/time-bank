// [v7.36.6] 安全解析时间戳，兼容多种格式
function parseTimestampSafe(ts) {
    if (typeof ts === 'number') return ts;
    if (typeof ts === 'string') {
        try {
            // 尝试ISO格式
            const dt = new Date(ts);
            const time = dt.getTime();
            return isNaN(time) ? 0 : time;
        } catch (e) {
            console.warn('[parseTimestampSafe] Failed to parse:', ts);
            return 0;
        }
    }
    return 0;
}

function buildTransactionIndex() {
    _transactionIndex = new Map();

        if (!_transactionIndex.has(tx.taskId)) {

    // [v7.36.6] 新增：习惯连胜健康检查（延迟5秒执行，避免阻塞启动）
    setTimeout(async () => {
        console.log('[HealthCheck] Verifying habit streak consistency...');
        const issues = [];
        
        for (const task of tasks) {
            if (!task.isHabit || !task.habitDetails) continue;
            
            // 检查 lastCompletionDate 是否与最近的 advancement 匹配
            const advTxs = transactions.filter(t => 
                t.taskId === task.id && t.isStreakAdvancement
            ).sort((a, b) => parseTimestampSafe(b.timestamp) - parseTimestampSafe(a.timestamp));
            
            if (advTxs.length > 0) {
                const latestAdv = advTxs[0];
                const expectedDate = getLocalDateString(latestAdv.timestamp);
                const actualDate = task.habitDetails.lastCompletionDate;
                
                if (expectedDate !== actualDate) {
                    issues.push({
                        taskName: task.name,
                        expected: expectedDate,
                        actual: actualDate,
                        streak: task.habitDetails.streak
                    });
                    
                    console.warn(`[HealthCheck] Mismatch detected for ${task.name}:`);
                    console.warn(`  Expected lastCompletionDate: ${expectedDate}`);
                    console.warn(`  Actual: ${actualDate}`);
                }
            } else if (task.habitDetails.streak > 0) {
                // 有连胜但没有 advancement 标记，也是异常
                issues.push({
                    taskName: task.name,
                    issue: 'streak without advancement markers',
                    streak: task.habitDetails.streak
                });
                console.warn(`[HealthCheck] ${task.name} has streak ${task.habitDetails.streak} but no advancement markers`);
            }
        }
        
        if (issues.length > 0) {
            console.warn(`[HealthCheck] Found ${issues.length} inconsistencies. Auto-repairing...`);
            
            for (const issue of issues) {
                const task = tasks.find(t => t.name === issue.taskName);
                if (task) {
                    console.log(`[HealthCheck] Repairing ${issue.taskName}...`);
                    rebuildHabitStreak(task);
                }
            }
            
            await saveData();
            
            // 同步到云端
            if (isLoggedIn()) {
                const syncPromises = issues
                    .map(issue => tasks.find(t => t.name === issue.taskName))
                    .filter(Boolean)
                    .map(task => DAL.saveTask(task).catch(err => {
                        console.error(`[HealthCheck] Sync failed for ${task.name}:`, err);
                    }));
                await Promise.all(syncPromises);
            }
            
            updateAllUI();
            showNotification('🔧 习惯数据已修复', `自动修复了 ${issues.length} 个任务的连胜状态`, 'info');
            console.log(`[HealthCheck] Auto-repair complete for ${issues.length} tasks`);
        } else {
            console.log('[HealthCheck] All habit streaks are consistent ✅');
        }
    }, 5000);

## ⚠️ 风险控制

### 已知风险

1. **性能回退**：
   - 风险：交易索引可能占用额外内存
   - 缓解：索引仅在首次调用 `rebuildHabitStreak()` 时构建，且可被GC回收
   - 监控：观察低端设备的内存使用情况

2. **云端同步冲突**：
   - 风险：多个设备同时修改同一任务的连胜状态
   - 缓解：依赖现有的云函数锁机制（`timebankTaskLock`）
   - 监控：检查云端日志中的冲突记录

3. **用户数据损坏**：
   - 风险：`rebuildHabitStreak()` 的逻辑bug导致错误标记
   - 缓解：保留完整的交易快照，支持回滚
   - 监控：收集用户反馈，准备hotfix分支

### 回滚计划

如果发现问题：

1. **立即措施**：
   ```javascript
   // 临时禁用健康检查
   // 注释掉 initApp() 中的 setTimeout 块
   ```

2. **代码回滚**：
   ```bash
   git revert HEAD
   git push
   ```

3. **数据恢复**：
   - 指导用户使用之前的备份文件重新导入
   - 或通过CloudBase控制台手动修复受影响的任务
